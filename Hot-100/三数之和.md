# 三数之和

给你一个整数数组 `nums` ，判断是否存在三元组 `[nums[i], nums[j], nums[k]]` 满足 `i != j`、`i != k` 且 `j != k` ，同时还满足 `nums[i] + nums[j] + nums[k] == 0` 。请你返回所有和为 `0` 且不重复的三元组。

**注意：**答案中不可以包含重复的三元组。

 

 

**示例 1：**

```
输入：nums = [-1,0,1,2,-1,-4]
输出：[[-1,-1,2],[-1,0,1]]
解释：
nums[0] + nums[1] + nums[2] = (-1) + 0 + 1 = 0 。
nums[1] + nums[2] + nums[4] = 0 + 1 + (-1) = 0 。
nums[0] + nums[3] + nums[4] = (-1) + 2 + (-1) = 0 。
不同的三元组是 [-1,0,1] 和 [-1,-1,2] 。
注意，输出的顺序和三元组的顺序并不重要。
```

**示例 2：**

```
输入：nums = [0,1,1]
输出：[]
解释：唯一可能的三元组和不为 0 。
```

**示例 3：**

```
输入：nums = [0,0,0]
输出：[[0,0,0]]
解释：唯一可能的三元组和为 0 。
```

 

**提示：**

- `3 <= nums.length <= 3000`
- `-105 <= nums[i] <= 105`

```csharp
// 使用双指针的思路，时间复杂度为 O(n**2)
// 将数组按照升序排列
// 在循环开始先做结束判断（当前数大于0，退出循环）
// 如果当前数与上一个数相等，continue（先做边界检查，再判断数值是否相等）
// 执行双指针逻辑，sum==0, Add(),去重，继续检查剩余数是否满足三数之和等于0
// sum < 0, L++  ; sum > 0, R--

public class Solution {
    public IList<IList<int>> ThreeSum(int[] nums) {
        var res = new List<IList<int>>();
        Array.Sort(nums);

        for (int i = 0; i < nums.Length - 2; i++) {
            // 如果最小数不为负，则三数之和不为0，退出循环
            if (nums[i] > 0) break;

            // 题目要求不含重复三元组
            // 去重操作
            if (i > 0 && nums[i] == nums[i - 1]) continue;

            // 双指针
            int L = i + 1;
            int R = nums.Length - 1;

            while (L < R) {
                int sum = nums[i] + nums[L] + nums[R];

                if (sum == 0) {
                    res.Add(new List<int> {nums[i], nums[L], nums[R]});

                    // 查看是否还有其他三元组组合
                    while (L < R && nums[L] == nums[L + 1]) L++;
                    while (L < R && nums[R] == nums[R - 1]) R--;

                    L++;
                    R--;
                } else if (sum < 0) {
                    L++;
                } else {
                    R--;
                }
            }
        }
        return res;
    }
}
```

```csharp
public class Solution {
    public IList<IList<int>> ThreeSum(int[] nums) {
        // 先进行排序
        // 设置一个指针遍历所有数
        // 设置一对左右指针，相向遍历
        // 满足要求，添加到 ans
        var ans = new List<IList<int>>();

        Array.Sort(nums);

        for (int i = 0; i < nums.Length; i++) {
            // 如果最小值都大于 0，则不存在符合要求的三元组
            if (nums[i] > 0) break;

            // 先做范围判断，防止越界
            // 如果数重复，继续移动到后一个数
            if (i > 0 && nums[i] == nums[i - 1]) continue;

            int l = i + 1, r = nums.Length - 1;

            while (l < r) {
                int sum = nums[i] + nums[l] + nums[r];

                // 如果和为0 将当前三元组添加到ans中
                // 去重，继续查找是否存在合适的其他三元组

                // 如果和 >0 nums[r]偏大，右边界左移
                // 反之 左边界右移
                if (sum == 0) {
                    ans.Add(new List<int> {nums[i], nums[l], nums[r]});
                    while (l < r && nums[l] == nums[l + 1]) l++;
                    while (l < r && nums[r] == nums[r - 1]) r--;
                    l++;
                    r--;
                } else if (sum < 0) {
                    l++;
                } else {
                    r--;
                }
            }
        }

        return ans;
    }
}
```
